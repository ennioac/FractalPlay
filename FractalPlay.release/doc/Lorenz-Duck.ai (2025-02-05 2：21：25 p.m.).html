Certainly! Below is a minimal Python program that computes the Lorenz attractor coordinates without any graphical output. It simply prints the x, y, z coordinates.

python
import numpy as np

def lorenz_attractor(sigma, beta, rho, dt, steps):
    x, y, z = 1.0, 1.0, 1.0  # Initial conditions
    for _ in range(steps):
        dx = sigma * (y - x) * dt
        dy = (x * (rho - z) - y) * dt
        dz = (x * y - beta * z) * dt
        x += dx
        y += dy
        z += dz
        print(x, y, z)

# Parameters
sigma = 10.0
beta = 8/3
rho = 28.0
dt = 0.01
steps = 1000

lorenz_attractor(sigma, beta, rho, dt, steps)
